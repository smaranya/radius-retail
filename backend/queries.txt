CREATE TABLE Users(
    UID VARCHAR(8) PRIMARY KEY, 
    Name VARCHAR(20) NOT NULL, 
    Phone VARCHAR(15) NOT NULL UNIQUE, 
    Password VARCHAR(20) NOT NULL, 
    Location POINT, 
    isChatting BOOLEAN NOT NULL
);

CREATE TABLE Product(
  PID VARCHAR(8) PRIMARY KEY,
  Category VARCHAR(15),
  Name VARCHAR(20),
  isPackaged BOOLEAN NOT NULL
);

CREATE TABLE Sale (
  SID VARCHAR(8) PRIMARY KEY,
  PID VARCHAR(8),
  Date DATE NOT NULL,
  UID VARCHAR(8),
  Quantity VARCHAR(6) NOT NULL,
  Quality INT NOT NULL,
  FOREIGN KEY (PID) 
    REFERENCES Product(PID)
    ON DELETE CASCADE,
  FOREIGN KEY (UID) 
    REFERENCES Users(UID)
    ON DELETE CASCADE
);
